   // create a card with the deedline 
@POST
@Path("/create/{deedline}") // Corrected path parameter syntax
public Response createCard(Card card, @PathParam("deedline") Date deedline) {
    User user = em.find(User.class, card.getReporterId());
    if (user == null) {
        return Response.status(Response.Status.NOT_FOUND).entity("User not Found").build();
    }
    card.setCreationDate(new Date());
    card.setDeedline(deedline);
    em.persist(card);
    notificationService.sendReminderNotification(card, 1); // One day before the deadline
    return Response.status(Response.Status.CREATED).entity(card).build();
} 




    @POST
    @Path("/{cardId}/deadline/{deadline}")
    public Response setDeadline(@PathParam("cardId") int cardId, @PathParam("deadline") Date deadline) {
        Card card = em.find(Card.class, cardId);
        if (card == null) {
            return Response.status(Response.Status.NOT_FOUND).build();
        }
        card.setDeadline(deadline);
        em.merge(card);
        notificationService.sendReminderNotification(card, 1); // One day before the deadline
        return Response.status(Response.Status.OK).build();
    }
